class Solution:
    def longestPalindrome(self, s: str) -> int:
        if s is None or len(s)==0:
            return 0
        Set=set()
        count=0
        for i in range(len(s)):
            c=s[i]
            if c in Set:
                count=count+2
                Set.remove(c)
            else:
                Set.add(c)
        if len(Set)>0:
            count=count+1
        return count



TC-O(n)
SC-O(1)
